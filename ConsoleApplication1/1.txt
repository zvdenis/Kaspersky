using System;
using System.Collections.Generic;

namespace ConsoleApplication1
{
    internal class Program
    {
        private static HashSet<string> hexSet = new HashSet<string>();

        public static bool isHex(string str)
        {
            if (str.Length != 8)
                return false;
            for (int i = 0; i < str.Length; i++)
            {
                if ((str[i] < '0' || '9' < str[i]) && (str[i] < 'A' || 'F' < str[i]))
                    return false;
            }

            return true;
        }
        
        public static bool IsUniqueBase(string str)
        {
            string[] parts = str.Split('/');
            string hex = "";
            int count = 0;
            for (int i = 0; i < parts.Length; i++)
            {
                if (isHex(parts[i]))
                {
                    hex = parts[i];
                    count++;
                }
            }

            if (count == 1)
            {
                if (hexSet.Contains(hex))
                {
                    return false;
                }

                hexSet.Add(hex);
                return true;
            }

            return false;
        }
        
        public static void Main()
        {
            string[] urls = Console.ReadLine().Split(' ');
            int count = 0; 
            foreach (var url in urls)
            {
                if (IsUniqueBase(url))
                {
                    count++;
                }
            }
            Console.WriteLine(count);
        }
    }
}
